SELECT P.PRODUCT_ID, PRODUCT_NAME
FROM PRODUCT P
JOIN (
    SELECT PRODUCT_ID 
    FROM SALES 
    GROUP BY PRODUCT_ID 
    HAVING MAX(SALE_DATE) BETWEEN TO_DATE('2019-01-01') AND TO_DATE('2019-03-31')
    AND MIN(SALE_DATE) BETWEEN TO_DATE('2019-01-01') AND TO_DATE('2019-03-31')
    ) S
ON P.PRODUCT_ID = S.PRODUCT_ID
